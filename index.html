🔧 配置 Supabase 云数据库
首次使用需要配置一次，数据会安全地同步到云端，任何设备登录都能访问。
1
打开 supabase.com 注册免费账号
2
点 New project，区域选 Northeast Asia (Tokyo)，设置密码
3
进入 Settings → API Keys，复制 Project URL 和 anon key
4
进入 SQL Editor，点下方「复制 SQL」按钮，粘贴后点 Run 运行
create table if not exists fhr_data(id text primary key, user_id uuid references auth.users, data jsonb, updated_at timestamptz default now());
alter table fhr_data enable row level security;
create policy "own" on fhr_data for all using(auth.uid()=user_id);
insert into storage.buckets(id,name,public) values('health-files','health-files',false) on conflict do nothing;
create policy "own_files" on storage.objects for all using(bucket_id='health-files' and auth.uid()::text=split_part(name,'/',1));
